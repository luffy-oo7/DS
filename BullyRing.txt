DynamicBully.java
import java.util.Scanner;

public class DynamicBully 
{
    static boolean[] active;
    static int totalNodes;

    public static void main(String[] args) 
    {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter total number of nodes: ");
        totalNodes = sc.nextInt();

        active = new boolean[totalNodes + 1];

        // Make all nodes active initially
        for (int i = 1; i <= totalNodes; i++) 
        {
            active[i] = true;
        }

        System.out.print("Enter the node that fails (1 to " + totalNodes + "): ");
        int failed = sc.nextInt();
        active[failed] = false;

        System.out.print("Enter the node that detects the failure and starts election: ");
        int starter = sc.nextInt();

        System.out.println("\n--- Election Process Started ---");

        startElection(starter);

        sc.close();
    }

    static void startElection(int starter) 
    {
        boolean higherNodeResponded = false;

        for (int i = starter + 1; i <= totalNodes; i++) 
        {
            if (active[i]) 
            {
                System.out.println("Node " + starter + " sends ELECTION to Node " + i);
                System.out.println("Node " + i + " responds OK to Node " + starter);

                higherNodeResponded = true;

                // Higher node takes over the election
                startElection(i);

                return; // Stop current node’s election
            }
        }

        // If no higher node responded → current node becomes leader
        if (!higherNodeResponded) 
        {
            System.out.println("\n*** Node " + starter + " IS THE NEW LEADER ***");

            for (int i = 1; i <= totalNodes; i++) 
            {
                if (active[i] && i != starter) 
                {
                    System.out.println("Node " + starter + " sends LEADER message to Node " + i);
                }
            }
        }
    }
}

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Scanner;

class RingNode 
{
    private int id;
    private boolean isActive;
    private int leaderId;
    private RingNode nextNode;

    public RingNode(int id) 
    {
        this.id = id;
        this.isActive = true;
    }

    public void setNextNode(RingNode nextNode) 
    {
        this.nextNode = nextNode;
    }

    public void fail() 
    {
        this.isActive = false;
        System.out.println("Node " + id + " has failed.");
    }

    public void startElection() 
    {
        if (!isActive) return;

        List<Integer> activeIds = new ArrayList<>();
        activeIds.add(this.id);

        System.out.println("[Node " + id + "] Starting election. Passing list: " + activeIds);

        nextNode.receiveElectionMessage(activeIds);
    }

    public void receiveElectionMessage(List<Integer> activeIds) 
    {
        if (!isActive) 
        {
            // Dead node passes message forward
            nextNode.receiveElectionMessage(activeIds);
            return;
        }

        if (activeIds.contains(this.id)) 
        {
            // Message completed full circle
            int newLeader = Collections.max(activeIds);

            System.out.println("\n--- Election Over. Node " + newLeader + " is the Leader ---");

            announceCoordinator(newLeader);
        } 
        else 
        {
            // Add ID and forward
            activeIds.add(this.id);

            System.out.println("[Node " + id + "] Received list. Adding self and passing: " + activeIds);

            nextNode.receiveElectionMessage(activeIds);
        }
    }

    public void announceCoordinator(int newLeaderId) 
    {
        if (this.leaderId == newLeaderId) 
        {
            return; // Stop when full circle is complete
        }

        this.leaderId = newLeaderId;

        System.out.println("[Node " + id + "] Acknowledged new Leader: " + leaderId);

        if (nextNode != null) 
        {
            nextNode.announceCoordinator(newLeaderId);
        }
    }
}

public class RingSimulation 
{
    public static void main(String[] args) 
    {
        Scanner scanner = new Scanner(System.in);

        // 1. Input total nodes
        System.out.print("Enter total number of nodes: ");
        int totalNodes = scanner.nextInt();

        RingNode[] nodes = new RingNode[totalNodes];

        for (int i = 0; i < totalNodes; i++) 
        {
            nodes[i] = new RingNode(i + 1);
        }

        // 2. Create ring
        for (int i = 0; i < totalNodes; i++) 
        {
            nodes[i].setNextNode(nodes[(i + 1) % totalNodes]);
        }

        // 3. Fail a node
        System.out.print("Enter the ID of the node that fails (1 to " + totalNodes + "): ");
        int failedNodeId = scanner.nextInt();

        nodes[failedNodeId - 1].fail();

        // 4. Start election
        System.out.print("Enter the ID of the node that starts the election (1 to " + totalNodes + "): ");
        int startNodeId = scanner.nextInt();

        nodes[startNodeId - 1].startElection();

        scanner.close();
    }
}
