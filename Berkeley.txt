server.py
from functools import reduce
from dateutil import parser
import threading
import datetime
import socket
import time

client_data = {}

def startReceivingClockTime(connector, address):
    while True:
        clock_time_string = connector.recv(1024).decode()
        clock_time = parser.parse(clock_time_string)
        clock_time_diff = datetime.datetime.now() - clock_time

        client_data[address] = {
            "clock_time": clock_time,
            "time_difference": clock_time_diff,
            "connector": connector
        }

        print("Client Data updated:", address)
        time.sleep(5)

def startConnecting(master_server):
    while True:
        conn, addr = master_server.accept()
        address = str(addr[0]) + ":" + str(addr[1])
        print(address + " connected")

        threading.Thread(
            target=startReceivingClockTime,
            args=(conn, address)
        ).start()

def getAverageClockDiff():
    diffs = [client['time_difference'] for client in client_data.values()]
    avg = sum(diffs, datetime.timedelta(0)) / len(diffs)
    return avg

def synchronizeAllClocks():
    while True:
        print("\nNew synchronization cycle")
        print("Clients:", len(client_data))

        if len(client_data) > 0:
            avg_diff = getAverageClockDiff()

            for client in client_data.values():
                try:
                    new_time = datetime.datetime.now() + avg_diff
                    client['connector'].send(str(new_time).encode())
                except:
                    print("Error sending time")

        time.sleep(5)

def initiateClockServer(port=8080):
    server = socket.socket()
    server.bind(('', port))
    server.listen(10)

    print("Server started...\n")

    threading.Thread(target=startConnecting, args=(server,)).start()
    threading.Thread(target=synchronizeAllClocks).start()

if __name__ == '__main__':
    initiateClockServer()



client.py
from dateutil import parser
import threading
import datetime
import socket
import time

def startSendingTime(client):
    while True:
        client.send(str(datetime.datetime.now()).encode())
        print("Time sent")
        time.sleep(5)

def startReceivingTime(client):
    while True:
        new_time = parser.parse(client.recv(1024).decode())
        print("Synchronized time:", new_time)

def initiateClient(port=8080):
    client = socket.socket()
    client.connect(('127.0.0.1', port))

    threading.Thread(target=startSendingTime, args=(client,)).start()
    threading.Thread(target=startReceivingTime, args=(client,)).start()

if __name__ == '__main__':
    initiateClient()
