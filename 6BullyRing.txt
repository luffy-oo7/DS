DynamicBully.java
import java.util.Scanner;
public class DynamicBully {
    static boolean[] active;
    static int totalNodes;
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter total number of nodes: ");
        totalNodes = sc.nextInt();
        active = new boolean[totalNodes + 1];
        for (int i = 1; i <= totalNodes; i++) {
            active[i] = true;
        }
        System.out.print("Enter failed node (1 to " + totalNodes + "): ");
        int failed = sc.nextInt();
        active[failed] = false;
        System.out.print("Enter node that starts election: ");
        int starter = sc.nextInt();
        System.out.println("\n--- Election Process Started ---\n");
        startElection(starter);
        sc.close();
    }
    static void startElection(int starter) {
        boolean higherFound = false;
        for (int i = starter + 1; i <= totalNodes; i++) {
            if (active[i]) {
                System.out.println("Node " + starter + " -> ELECTION -> Node " + i);
                System.out.println("Node " + i + " -> OK -> Node " + starter);
                higherFound = true;
                startElection(i); // higher node takes over
                return;
            }
        }
        if (!higherFound) {
            System.out.println("\n*** Node " + starter + " is LEADER ***\n");
            for (int i = 1; i <= totalNodes; i++) {
                if (active[i] && i != starter) {
                    System.out.println("Node " + starter + " sends COORDINATOR to Node " + i);
                }
            }
        }
    }
}



RingSimulation.java
import java.util.*;
class RingNode {
    private int id;
    private boolean isActive = true;
    private int leaderId = -1;
    private RingNode nextNode;
    public RingNode(int id) {
        this.id = id;
    }
    public void setNextNode(RingNode nextNode) {
        this.nextNode = nextNode;
    }
    public void fail() {
        isActive = false;
        System.out.println("Node " + id + " has failed.");
    }
    public void startElection() {
        if (!isActive) return;
        List<Integer> list = new ArrayList<>();
        list.add(id);
        System.out.println("\nNode " + id + " starts election: " + list);
        nextNode.receiveElectionMessage(list);
    }
    public void receiveElectionMessage(List<Integer> list) {
        if (!isActive) {
            nextNode.receiveElectionMessage(list);
            return;
        }
        if (list.contains(id)) {
            int leader = Collections.max(list);
            System.out.println("\n--- Leader elected: Node " + leader + " ---\n");
            announceLeader(leader);
        } else {
            list.add(id);
            System.out.println("Node " + id + " passing list: " + list);
            nextNode.receiveElectionMessage(list);
        }
    }
    public void announceLeader(int leader) {
        if (this.leaderId == leader) return;
        this.leaderId = leader;
        System.out.println("Node " + id + " acknowledges Leader " + leader);
        nextNode.announceLeader(leader);
    }
}
public class RingSimulation {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter number of nodes: ");
        int n = sc.nextInt();
        RingNode[] nodes = new RingNode[n];
        for (int i = 0; i < n; i++) {
            nodes[i] = new RingNode(i + 1);
        }
        for (int i = 0; i < n; i++) {
            nodes[i].setNextNode(nodes[(i + 1) % n]);
        }
        System.out.print("Enter failed node: ");
        int failed = sc.nextInt();
        nodes[failed - 1].fail();
        System.out.print("Enter starter node: ");
        int start = sc.nextInt();
        nodes[start - 1].startElection();
        sc.close();
    }
}
